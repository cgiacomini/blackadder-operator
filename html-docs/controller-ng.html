 13        """
 14        Initializes the ChaosController by setting up logging, configuring the Kubernetes client,

 16        """
 17        self.setup_logging()
 18        try:
 19            self.config = pykube.KubeConfig.from_env()
 20            self.api = pykube.HTTPClient(self.config)
 21            self.ChaosAgent = pykube.object_factory(self.api, "blackadder.io/v1beta1", "ChaosAgent")
 22            self.agent = self.get_agent()
 23            if not self.agent:
 24                self.logger.error("No ChaosAgent found, exiting")
 25                sys.exit(1)
 26            self.exclude_namespaces = self.agent.config.excludedNamespaces
 27            self.pause_duration = int(self.agent.config.pauseDuration)
 28        except Exception as e:
 29            self.logger.error(f"Error initializing ChaosController: {e}")
 30            sys.exit(1)

 43    def get_agent(self):
 44        """
 45        Retrieves the ChaosAgent configuration from the Kubernetes API server.
 46
 47        Returns:
 48            agent (pykube.objects): The ChaosAgent object with its configuration.
 49        """
 50        try:
 51            agents = list(self.ChaosAgent.objects(self.api, namespace=pykube.all))
 52            if not agents:

 60            sys.exit(1)

 81    def randomly_kill_pods(self, pods, tolerance, eagerness):
 82        """
 83        Randomly kills pods based on the specified tolerance and eagerness.
 84
 85        Args:
 86            pods (list): A list of Pod objects.
 87            tolerance (int): The minimum number of pods to keep alive.
 88            eagerness (int): The percentage chance to kill each pod.
 89        """
 90        try:
 91            if len(pods) < tolerance:
 92                self.logger.info(f"Skipping pod killing, only {len(pods)} pods found")
 93                return
 94            # Randomly select PODs to kill
 95            pods_to_kill = []
 96            for pod in pods:
 97                if random.randint(1, 100) <= eagerness:
 98                    pods_to_kill.append(pod)
 99            
100            # Ensure we do not kill more pods than allowed by tolerance
101            if len(pods) - len(pods_to_kill) < tolerance:
102                self.logger.info(f"Skipping pod killing, would violate tolerance with {len(pods_to_kill)} pods to kill")
103                return
104            
105            # Kill selected PODs
106            self.logger.info(f"Attempting to kill {len(pods_to_kill)} pods out of {len(pods)} available pods")
107            for pod in pods_to_kill:
108                try:
109                    self.logger.info(f"Killing pod {pod.namespace}/{pod.name}")
110                    pod.delete()
111                    self.logger.info(f"Killed pod {pod.namespace}/{pod.name}")
112                except Exception as e:
113                    self.logger.error(f"Error killing pod {pod.namespace}/{pod.name}: {e}")
114        except Exception as e:
115            self.logger.error(f"Error killing pods: {e}")

117    def randomly_scale_deployments(self, deployments, eagerness):
118        """
119        Randomly scales deployments based on the specified eagerness.
120
121        Args:
122            deployments (list): A list of Deployment objects.
123            eagerness (int): The percentage chance to scale each deployment.
124        """
125        for deployment in deployments:
126            if random.randint(0, 100) < eagerness:
127                try:
128                    current_replicas = deployment.obj['spec']['replicas']
129                    #  Scale deployment to double the current number of replicas, up to a maximum of 128
130                    if current_replicas < 128:
131                        new_replicas = min(current_replicas * 2, 128)
132                    else:
133                        new_replicas = current_replicas  # No change if already 128 or more
134
135                    # Update the deployment with the new number of replicas
136                    deployment.obj['spec']['replicas'] = new_replicas
137                    deployment.update()
138                    self.logger.info(f"Scaled {deployment.namespace}/{deployment.name} to {new_replicas}")
139                except Exception as e:
140                    self.logger.error(f"Error scaling deployment {deployment.namespace}/{deployment.name}: {e}")

142    def randomly_write_configmaps(self, configmaps, eagerness):
143        """
144        Randomly modifies the data in Kubernetes ConfigMaps based on a specified eagerness level.
145
146        Args:
147            configmaps (list): A list of ConfigMap objects.
148            eagerness (int): The percentage chance to modify each ConfigMap.
149        """
150        for cm in configmaps:
151            self.logger.info(f"Checking {cm.namespace}/{cm.name}")
152            
153            # Check if the ConfigMap is immutable
154            if cm.obj.get("immutable"):
155                self.logger.info(f"Skipping immutable ConfigMap {cm.namespace}/{cm.name}")
156                continue
157            
158            # Randomly decide whether to modify the ConfigMap based on eagerness
159            if random.randint(0, 100) < eagerness:
160                try:
161                    # Replace each value in the data section with Lorem Ipsum text
162                    for k, v in cm.obj["data"].items():
163                        cm.obj["data"][k] = lorem.paragraph()
164                    cm.update()
165                    self.logger.info(f"Modified ConfigMap {cm.namespace}/{cm.name} with Lorem Ipsum text")
166                except Exception as e:
167                    self.logger.error(f"Error modifying ConfigMap {cm.namespace}/{cm.name}: {e}")
168                    
169    def run(self):
170        """
171        Runs the main loop of the ChaosController, periodically listing objects and performing chaos actions.
172        """
173        while True:
174            try:
175                pods = self.list_objects(Pod)
176                deployments = self.list_objects(Deployment)
177                if pods:
178                    self.randomly_kill_pods(pods, tolerance=self.agent.config.tolerance, eagerness=self.agent.config.eagerness)
179                if deployments:
180                    self.randomly_scale_deployments(deployments, eagerness=self.agent.config.eagerness)
181            except Exception as e:
182                self.logger.error(f"Error in main loop: {e}")
183
184            time.sleep(self.pause_duration)
